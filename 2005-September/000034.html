+    def testMakeDocumentId(self):
+        handle, filename = mkstemp()
+        f = os.fdopen(handle, 'wb')

+        f.write(data)
+        f.close()
+        self.assertEquals(makeDocumentId(filename), sha.sha(data).hexdigest())
